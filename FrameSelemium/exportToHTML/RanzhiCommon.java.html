43        * 然之的切换语言的业务 
44        */ 
45       String changeLanguage(String lang) throws InterruptedException {
46           WebDriver driver = this.baseDriver;
47           // 点击语言按钮 
48           driver.findElement(By.cssSelector("#langs > button")).click();
49           Thread.sleep(500);
50           // 用Css Selector 选择 语言
51           String langNumber = "";
52           switch (lang) { 
53               case "en":
54                   langNumber = "3";
55                   break;
56               case "zh_CN":
57                   langNumber = "1";
58                   break;
59               case "zh_TW":
60                   langNumber = "2";
61                   break;
62           } 
63           String langSelect = "#langs > ul > li:nth-child(" + langNumber + ") > a";
64           driver.findElement(By.cssSelector(langSelect)).click();
65           // 浏览器需要刷新，等待2秒钟 
66           Thread.sleep(2000);
67    
68           // 找到语言按钮的文字，返回给调用者
69           String language = 
70                   driver.findElement(By.cssSelector("#langs > button")).getText();
71    
72           return language;
73       }

99       void logOut(String lang) throws InterruptedException {
100          WebDriver driver = this.baseDriver;
101   
102          // 点头像（左下角） 
103          driver.findElement(By.cssSelector("#start > div")).click();
104          Thread.sleep(500);
105   
106          // 在弹出的上下文菜单，选择文字(根据语言）
107          // 1. 中文简体：退出 
108          // 2. 中文繁体：退出 
109          // 3. English：Logout
110          // 用Switch来写 
111          String exitText = "";
112          switch (lang) { 
113              case "en":
114                  exitText = "Logout";
115                  break;
116              case "zh_CN":
117              case "zh_TW":
118                  exitText = "退出";
119                  break;
120          } 
121          driver.findElement(By.linkText(exitText)).click();
122          Thread.sleep(2000);
123   
124      }

163          WebDriver driver = this.baseDriver;
164          // 添加成员按钮 在 紫禁城中 
165          WebElement frameAdmin = driver.findElement(By.cssSelector(this.ADMIN_FRAME_SELECTOR));
166          driver.switchTo().frame(frameAdmin);
167   
168          // 点击添加成员按钮 
169          driver.findElement(By.cssSelector(this.ADMIN_ADD_MEMBER_BUTTON_SELECTOR)).click();
170          Thread.sleep(1000);
171   
172          // 办完事儿后，退出 紫禁城 
173          // 必须 进退 成对出现 
174          driver.switchTo().defaultContent();
175      }

177      void addMemberData(Member member) throws InterruptedException {
178          WebDriver driver = this.baseDriver;
179          // 添加成员按钮 在 紫禁城中 
180          WebElement frameAdmin = driver.findElement(By.cssSelector(this.ADMIN_FRAME_SELECTOR));
181          driver.switchTo().frame(frameAdmin);
182   
183          // 开始填表单 
184          // 用户名 
185          WebElement elementAccount = driver.findElement(By.cssSelector(this.ADMIN_ADD_MEMBER_EDIT_ACCOUNT));
186          elementAccount.clear();
187          elementAccount.sendKeys(member.getAccount());
188   
189          // 真实姓名 
190          WebElement elementRealName = driver.findElement(By.cssSelector(this.ADMIN_ADD_MEMBER_EDIT_REALNAME));
191          elementRealName.clear();
192          elementRealName.sendKeys(member.getRealName());
193   
194          // 性别 
195          if (member.getGender() == Member.Gender.Female) {
196              driver.findElement(By.cssSelector(this.ADMIN_ADD_MEMBER_EDIT_GENDER_FEMALE)).click();
197          } else { 
198              driver.findElement(By.cssSelector(this.ADMIN_ADD_MEMBER_EDIT_GENDER_MALE)).click();
199          } 
200          // 部门 
201          WebElement elementDepartment = driver.findElement(By.cssSelector(this.ADMIN_ADD_MEMBER_EDIT_DEPARTMENT));
202          Select objectDepartment = new Select(elementDepartment);
203          objectDepartment.selectByIndex(member.getDept());
204   
205          // 角色 
206          // 是个Select元素，不可以直接操作
207          WebElement elementRole = driver.findElement(By.cssSelector(this.ADMIN_ADD_MEMBER_EDIT_ROLE));
208          // elementRole 不可以直接操作
209          // 把 elementRole 构造成一个 Select() 对象
210          Select objectRole = new Select(elementRole);
211          objectRole.selectByIndex(member.getRole());
212   
213   
214          // 密码 
215          WebElement elementPassword = driver.findElement(By.cssSelector(this.ADMIN_ADD_MEMBER_EDIT_PASSWORD));
216          elementPassword.clear();
217          elementPassword.sendKeys(member.getPassword());
218   
219          // 重复密码 
220          WebElement elementPasswordRepeat = driver.findElement(By.cssSelector(this.ADMIN_ADD_MEMBER_EDIT_PASSWORD_REPEAT));
221          elementPasswordRepeat.clear();
222          elementPasswordRepeat.sendKeys(member.getPassword());
223   
224          // 邮箱 
225          WebElement elementEmail = driver.findElement(By.cssSelector(this.ADMIN_ADD_MEMBER_EDIT_EMAIL));
226          elementEmail.clear();
227          elementEmail.sendKeys(member.getEmail());
228   
229          // 点击保存按钮 
230          driver.findElement(By.cssSelector(this.ADMIN_ADD_MEMBER_SUBMIT_SELETOR)).click();
231   
232          Thread.sleep(2000);
233   
234          driver.switchTo().defaultContent();
235   
236   
237      }
