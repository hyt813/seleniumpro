24       /** 
25        * 测试登录 
26        * 需要用 @Test 注解进行标注 
27        * 这样 不用main()方法 便可直接运行测试 
28        * @throws InterruptedException 
29        */ 
30       @Test 
31       public void test02LogIn() throws InterruptedException {
32           WebDriver driver = this.baseDriver;
33           driver.get(this.baseUrl);
34           Thread.sleep(2000);
35    
36           driver.findElement(By.id("account")).sendKeys("admin");
37           driver.findElement(By.id("password")).sendKeys("123456");
38           driver.findElement(By.id("submit")).click();
39           Thread.sleep(2000);
40           String expectedUrl = "http://localhost/ranzhi/www/sys/index.html";
41           Assert.assertEquals(driver.getCurrentUrl(), expectedUrl);
42    
43       } 
44    
45       /** 
46        * 测试切换语言 
47        * 把系统语言切换成 English 
48        * 然后查询 语言的按钮 是不是变成了 English 
49        * @throws InterruptedException 
50        */ 
51       @Test 
52       public void test01ChangeLanguage() throws InterruptedException {
53           WebDriver driver = this.baseDriver;
54           driver.get(this.baseUrl);
55           Thread.sleep(2000);
56    
57           driver.findElement(By.cssSelector("#langs > button")).click();
58           Thread.sleep(500);
59           driver.findElement(By.cssSelector("#langs > ul > li:nth-child(3) > a")).click();
60           Thread.sleep(2000);
61    
62           String expected_language = "English";
63           String actual_language = 
64                   driver.findElement(By.cssSelector("#langs > button")).getText();
65    
66           Assert.assertEquals(actual_language, expected_language);
67    
68       }
