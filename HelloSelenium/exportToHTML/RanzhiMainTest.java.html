10   /** 
11    * Created by Linty on 1/8/2017. 
12    * 使用 @BeforeTest 和 @AfterTest 进行测试框架操作 
13    * 如果直接运行整个测试，运行步骤如下 
14    * 首先运行 @BeforeTest 
15    * 然后运行 test01ChangeLanguage 
16    * 接着运行 test02LogIn 
17    * 最后运行 @AfterTest 
18    */ 
19   public class RanzhiMainTest { 
20       // 声明两个全局变量，每个方法都可以用下面的变量。
21       private WebDriver baseDriver = null;
22       private String baseUrl = null;
23    
24       /** 
25        * 测试登录 
26        * 需要用 @Test 注解进行标注 
27        * 这样 不用main()方法 便可直接运行测试 
28        * @throws InterruptedException 
29        */ 
30       @Test 
31       public void test02LogIn() throws InterruptedException {
32           WebDriver driver = this.baseDriver;
33           driver.get(this.baseUrl);
34           Thread.sleep(2000);
35    
36           driver.findElement(By.id("account")).sendKeys("admin");
37           driver.findElement(By.id("password")).sendKeys("123456");
38           driver.findElement(By.id("submit")).click();
39           // 点击登录按钮后，需要等待浏览器刷新
40           Thread.sleep(2000);
41           String expectedUrl = "http://localhost/ranzhi/www/sys/index.html";
42           // driver.getCurrentUrl() -- 获取当前的浏览器URL
43           Assert.assertEquals(driver.getCurrentUrl(), expectedUrl);
44    
45       } 
46    
47       /** 
48        * 测试切换语言 
49        * 把系统语言切换成 English 
50        * 然后查询 语言的按钮 是不是变成了 English 
51        * @throws InterruptedException 
52        */ 
53       @Test 
54       public void test01ChangeLanguage() throws InterruptedException {
55           WebDriver driver = this.baseDriver;
56           driver.get(this.baseUrl);
57           Thread.sleep(2000);
58    
59           // 点击语言按钮 
60           driver.findElement(By.cssSelector("#langs > button")).click();
61           Thread.sleep(500);
62           // 用Css Selector 选择 英文
63           driver.findElement(By.cssSelector("#langs > ul > li:nth-child(3) > a")).click();
64           // 浏览器需要刷新，等待2秒钟 
65           Thread.sleep(2000);
66    
67           // 检查按钮上的字是不是变成了 English
68           String expected_language = "English";
69           String actual_language = 
70                   driver.findElement(By.cssSelector("#langs > button")).getText();
71    
72           Assert.assertEquals(actual_language, expected_language);
73    
74       } 
75    
76       /** 
77        * 测试前置条件 
78        * 在所有的测试开始前 执行一次 
79        */ 
80       @BeforeTest 
81       public void setUp() { 
82           this.baseDriver = new FirefoxDriver();
83           this.baseUrl = "http://localhost/ranzhi/www/";
84       } 
85    
86       /** 
87        * 测试清理操作 
88        * 在所有的测试结束后 执行一次 
89        */ 
90       @AfterTest 
91       public void tearDown(){ 
92           this.baseDriver.quit();
93       }
